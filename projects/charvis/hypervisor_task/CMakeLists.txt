cmake_minimum_required(VERSION 3.7.2)

project(hypervisor_task C)

set(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} -std=gnu99")

file(
    GLOB
        deps
        src/*.c
)

list(SORT deps)

add_library(hypervisor STATIC EXCLUDE_FROM_ALL ${deps})
target_include_directories(hypervisor PUBLIC include "include")
